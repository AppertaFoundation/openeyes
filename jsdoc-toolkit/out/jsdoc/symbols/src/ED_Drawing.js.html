 99 /*
100  * Chris Raettig's function for getting accurate mouse position in all browsers
101  *
102  * @param {Object} obj Object to get offset for, usually canvas object
103  * @returns {Object} x and y values of offset
104  */
105 ED.findOffset = function(obj)
106 {
107     var curleft = curtop = 0;
108     if (obj.offsetParent) {
109         do {
110             curleft += obj.offsetLeft;
111             curtop += obj.offsetTop;
112         } while (obj = obj.offsetParent);
113         return { x: curleft, y: curtop };
114     }
115 }

138 /**
139  * A Drawing consists of one canvas element displaying one or more doodles;
140  * Doodles are drawn in the 'doodle plane' consisting of a 1001 pixel square grid -500 to 500) with central origin, and negative Y upwards;
141  * Affine transforms are used to convert points in the doodle plane to the canvas plane, the plane of the canvas element;
142  * Each doodle contains additional transforms to handle individual position, rotation, and scale.
143  * 
144  * @class Drawing
145  * @property {Canvas} canvas A canvas element used to edit and display the drawing
146  * @property {Eye} eye Right or left eye (some doodles display differently according to side)
147  * @property {Context} context The 2d context of the canvas element
148  * @property {Array} doodleArray Array of doodles in the drawing
149  * @property {AffineTransform} transform Transform converts doodle plane -> canvas plane
150  * @property {AffineTransform} inverseTransform Inverse transform converts canvas plane -> doodle plane
151  * @property {Doodle} selectedDoodle The currently selected doodle, null if no selection
152  * @property {Bool} mouseDown Flag indicating whether mouse is down in canvas
153  * @property {Mode} mode The current mouse dragging mode
154  * @property {Point} lastMousePosition Last position of mouse in canvas coordinates
155  * @property {Image} image Optional background image
156  * @property {Int} doubleClickMilliSeconds Duration of double click
157  * @property {Bool} newPointOnClick Flag indicating whether a mouse click will create a new PointInLine doodle
158  * @property {Bool} completeLine Flag indicating whether to draw an additional line to the first PointInLine doodle
159  * @property {Float} scale Scaling of transformation from canvas to doodle planes, preserving aspect ratio and maximising doodle plnae
160  * @property {Float} globalScaleFactor Factor used to scale all added doodles to this drawing, defaults to 1
161  * @param {Canvas} _canvas Canvas element 
162  * @param {Eye} _eye Right or left eye
163  * @param {String} _IDSuffix String suffix to identify HTML elements related to this drawing
164  */

166 {
167 	// Properties
168 	this.canvas = _canvas;
169 	this.eye = _eye;
170 	this.IDSuffix = _IDSuffix;
171 	
172 	this.context = this.canvas.getContext('2d');
173 	this.doodleArray = new Array();
174 	this.transform = new ED.AffineTransform();
175 	this.inverseTransform = new ED.AffineTransform();
176 	this.selectedDoodle = null;
177 	this.mouseDown = false;
178     this.doubleClick = false;
179 	this.mode = ED.Mode.Move;
180 	this.lastMousePosition = new ED.Point(0, 0);
181     this.doubleClickMilliSeconds = 250;
182     this.onLoadedHasRun = false;
183     this.newPointOnClick = false;
184     this.completeLine = false;
185     this.globalScaleFactor = 1;
186     
187     // Fit canvas making maximum use of doodle plane
188     if (this.canvas.width >= this.canvas.height)
189     {
190         this.scale = this.canvas.width/1001;
191     }
192     else
193     {
194         this.scale = this.canvas.height/1001;
195     }
196     
197     // Array of images to be preloaded (Add new images here)
198     this.imageArray = new Array();
199     this.imageArray['latticePattern'] = new Image();
200     this.imageArray['cryoPattern'] = new Image();
201     this.imageArray['antPVRPattern'] = new Image();
202     this.imageArray['laserPattern'] = new Image();
203     this.imageArray['fuchsPattern'] = new Image();
204     this.imageArray['pscPattern'] = new Image();
205     this.imageArray['meshworkPattern'] = new Image();
206     this.imageArray['newVesselPattern'] = new Image();
207     
208 	// Set transform to map from doodle to canvas plane
209 	this.transform.translate(this.canvas.width/2, this.canvas.height/2);
210 	this.transform.scale(this.scale, this.scale);
211 	
212 	// Set inverse transform to map the other way
213 	this.inverseTransform = this.transform.createInverse();
214 	
215 	// Initialise canvas context transform by calling clear() method	
216 	this.clear();
217 	
218 	// Get reference to button elements
219 	this.moveToFrontButton = document.getElementById('moveToFront' + this.IDSuffix);
220 	this.moveToBackButton = document.getElementById('moveToBack' + this.IDSuffix);
221 	this.deleteButton = document.getElementById('delete' + this.IDSuffix);
222 	this.lockButton = document.getElementById('lock' + this.IDSuffix);
223 	this.unlockButton = document.getElementById('unlock' + this.IDSuffix);
224     this.squiggleSpan = document.getElementById('squiggleSpan' + this.IDSuffix);
225     this.colourPreview = document.getElementById('colourPreview' + this.IDSuffix);
226     this.fillRadio = document.getElementById('fillRadio' + this.IDSuffix);
227     this.thickness = document.getElementById('thicknessSelect' + this.IDSuffix);
228     
229     // Add event listeners (NB within the event listener 'this' refers to the canvas, NOT the drawing instance)
230     var drawing = this;
231     
232     // Mouse listeners
233     this.canvas.addEventListener('mousedown', function(e) {
234                             var offset = ED.findOffset(this);
235                             var point = new ED.Point(e.pageX-offset.x,e.pageY-offset.y);
236                             drawing.mousedown(point);
237                              }, false);
238     
239     this.canvas.addEventListener('mouseup', function(e) { 
240                             var offset = ED.findOffset(this);
241                             var point = new ED.Point(e.pageX-offset.x,e.pageY-offset.y);
242                             drawing.mouseup(point); 
243                             }, false);
244     
245     this.canvas.addEventListener('mousemove', function(e) { 
246                             var offset = ED.findOffset(this);
247                             var point = new ED.Point(e.pageX-offset.x,e.pageY-offset.y);
248                             drawing.mousemove(point); 
249                             }, false);
250     
251     this.canvas.addEventListener('mouseout', function(e) { 
252                             var offset = ED.findOffset(this);
253                             var point = new ED.Point(e.pageX-offset.x,e.pageY-offset.y);
254                             drawing.mouseout(point); 
255                             }, false);
256     
257     // iOS listeners
258     this.canvas.addEventListener('touchstart', function(e) { 
259                             var point = new ED.Point(e.targetTouches[0].pageX - this.offsetLeft,e.targetTouches[0].pageY - this.offsetTop);
260                             e.preventDefault();
261                             drawing.mousedown(point); 
262                             }, false);
263     
264     this.canvas.addEventListener('touchend', function(e) { 
265                             var point = new ED.Point(e.targetTouches[0].pageX - this.offsetLeft,e.targetTouches[0].pageY - this.offsetTop);
266                             drawing.mouseup(point); 
267                             }, false);
268     
269     this.canvas.addEventListener('touchmove', function(e) { 
270                             var point = new ED.Point(e.targetTouches[0].pageX - this.offsetLeft,e.targetTouches[0].pageY - this.offsetTop);
271                             drawing.mousemove(point); 
272                             }, false);
273     
274     // Keyboard listener
275     window.addEventListener('keydown',function(e) {
276                             if (document.activeElement && document.activeElement.tagName == 'CANVAS') drawing.keydown(e);
277                             }, true);
278     
279     
280     // Stop browser stealing double click to select text
281     this.canvas.onselectstart = function () { return false; }
282 }

306 /**
307  * Checks all images are loaded then calls onLoaded function to proceed with other processing
308  */
309 ED.Drawing.prototype.checkAllLoaded = function()
310 {
311     // Set flag to check loading
312     var allLoaded = true;
313     
314     // Iterate through array loading each image, checking all are loaded
315     for (var key in this.imageArray)
316     {
317         var imageLoaded = false;
318         if (this.imageArray[key].width > 0) imageLoaded = true;
319         
320         // Check all are loaded
321         allLoaded = allLoaded && imageLoaded;
322     }
323     
324     // If all are loaded, proceed with onLoaded event (if defined)
325     if (allLoaded && typeof(this.onLoaded) == 'function')
326     {
327         if (!this.onLoadedHasRun)
328         {
329             this.onLoaded();
330             this.onLoadedHasRun = true;
331         }
332     }
333 }

335 /**
336  * Loads doodles from passed set into doodleArray
337  *
338  * @param {Set} _doodleSet Set of doodles from server
339  */
340 ED.Drawing.prototype.load = function(_doodleSet)
341 {
342 	// Iterate through set of doodles and load into doodle array
343 	for (var i = 0; i < _doodleSet.length; i++)
344 	{
345 		// Instantiate a new doodle object with parameters from doodle set
346 		this.doodleArray[i] = new ED[_doodleSet[i].subclass]
347 		(
348 			this,
349 			_doodleSet[i].originX,
350 			_doodleSet[i].originY,
351 			_doodleSet[i].apexX,
352 			_doodleSet[i].apexY,
353 			_doodleSet[i].scaleX,
354 			_doodleSet[i].scaleY,
355 			_doodleSet[i].arc,
356 			_doodleSet[i].rotation,
357 			_doodleSet[i].order
358 		);
359         
360         // Squiggle array
361         if (typeof(_doodleSet[i].squiggleArray) != 'undefined')
362         {
363             for (var j = 0; j < _doodleSet[i].squiggleArray.length; j++)
364             {
365                 // Get paramters and create squiggle
366                 var c = _doodleSet[i].squiggleArray[j].colour;
367                 var colour = new ED.Colour(c.red, c.green, c.blue, c.alpha);
368                 var thickness = _doodleSet[i].squiggleArray[j].thickness;
369                 var filled = _doodleSet[i].squiggleArray[j].filled;
370                 var squiggle = new ED.Squiggle(this.doodleArray[i], colour, thickness, filled);
371                 
372                 // Add points to squiggle and complete it
373                 var pointsArray = _doodleSet[i].squiggleArray[j].pointsArray;
374                 for (var k = 0; k < pointsArray.length; k++)
375                 {
376                     var point = new ED.Point(pointsArray[k].x, pointsArray[k].y);
377                     squiggle.addPoint(point);
378                 }
379                 squiggle.complete = true;
380                 
381                 // Add squiggle to doodle's squiggle array
382                 this.doodleArray[i].squiggleArray.push(squiggle);
383             }
384         }
385 	}
386 	
387 	// Sort array by order (puts back doodle first)
388 	this.doodleArray.sort(function(a,b){return a.order - b.order});
389 }

418 /**
419  * Draws all doodles for this drawing
420  */ 
421 ED.Drawing.prototype.drawAllDoodles = function()
422 {
423     // Draw any connecting lines
424 
425     var ctx = this.context;
426     ctx.beginPath();
427     var started = false;
428     var startPoint;
429     
430     for (var i = 0; i < this.doodleArray.length; i++)
431     {
432         if (this.doodleArray[i].isPointInLine)
433         {
434             // Start or draw line
435             if (!started)
436             {
437                 ctx.moveTo(this.doodleArray[i].originX, this.doodleArray[i].originY);
438                 started = true;
439                 startPoint = new ED.Point(this.doodleArray[i].originX, this.doodleArray[i].originY);
440             }
441             else
442             {
443                 ctx.lineTo(this.doodleArray[i].originX, this.doodleArray[i].originY);
444             }
445         }
446     }
447     
448     // Optionally add line to start
449     if (this.completeLine && typeof(startPoint) != 'undefined')
450     {
451         ctx.lineTo(startPoint.x, startPoint.y);
452     }
453     
454     // Draw lines
455     if (started)
456     {
457         ctx.lineWidth = 4;
458         ctx.strokeStyle = "rgba(20,20,20,1)";
459         ctx.stroke();
460     }
461 
462     
463 	// Draw doodles
464 	for (var i = 0; i < this.doodleArray.length; i++)
465 	{
466 		// Save context (draw method of each doodle may alter it)
467 		this.context.save();
468 		
469 		// Draw doodle
470 		this.doodleArray[i].draw();
471 		
472 		// Restore context
473 		this.context.restore();
474 	}
475 }

478 /**
479  * Responds to mouse down event in canvas, cycles through doodles from front to back.
480  * Selected doodle is first selectable doodle to have click within boundary path.
481  * Double clicking on a selected doodle promotes it to drawing mode (if is drawable)
482  *
483  * @event
484  * @param {Point} _point Coordinates of mouse in canvas plane
485  */  
486 ED.Drawing.prototype.mousedown = function(_point)
487 {
488 	// Set flag to indicate dragging can now take place
489 	this.mouseDown = true;
490     
491     // Detect double click
492     if (ED.recentClick) this.doubleClick = true;
493     ED.recentClick = true;
494     var t = setTimeout("ED.recentClick = false;",this.doubleClickMilliSeconds);
495     
496 	// Set flag to indicate success
497 	var found = false;
498 	this.selectedDoodle = null;
499 
500 	// Cycle through doodles from front to back doing hit test
501 	for (var i = this.doodleArray.length - 1; i > -1; i--)
502 	{
503 		if (!found)
504 		{
505 			// Save context (draw method of each doodle may alter it)
506 			this.context.save();
507 		
508 			// Successful hit test?
509 			if (this.doodleArray[i].draw(_point))
510 			{
511 				if (this.doodleArray[i].isSelectable)
512 				{
513                     // If double clicked, go into drawing mode
514                     if (this.doubleClick && this.doodleArray[i].isSelected && this.doodleArray[i].isDrawable)
515                     {
516                         this.doodleArray[i].isForDrawing = true;
517                     }
518                     
519 					this.doodleArray[i].isSelected = true;
520 					this.selectedDoodle = this.doodleArray[i];
521 					found = true;
522                     
523                     // If for drawing, mouse down starts a new squiggle
524                     if (!this.doubleClick && this.doodleArray[i].isForDrawing)
525                     {
526                         // Add new squiggle
527                         this.doodleArray[i].addSquiggle();
528                     }
529 				}
530 			}
531 			// Ensure that unselected doodles are marked as such
532 			else
533 			{
534 				this.doodleArray[i].isSelected = false;
535                 this.doodleArray[i].isForDrawing = false;
536 			}
537 			
538 			// Restore context
539 			this.context.restore();
540 		}
541 		else
542 		{
543 			this.doodleArray[i].isSelected = false;
544             this.doodleArray[i].isForDrawing = false;
545 		}
546 		
547 		// Ensure drag flagged is off for each doodle
548 		this.doodleArray[i].isBeingDragged = false;
549 	}
550     
551     
552     if (this.newPointOnClick && !found)
553     {
554         var mousePosDoodlePlane = this.inverseTransform.transformPoint(_point);
555         
556         var newPointInLine = this.addDoodle('PointInLine');
557         newPointInLine.originX = mousePosDoodlePlane.x;
558         newPointInLine.originY = mousePosDoodlePlane.y;
559     }
560 	
561 	// Repaint
562 	this.repaint();
563 }

565 /**
566  * Responds to mouse move event in canvas according to the drawing mode
567  *
568  * @event
569  * @param {Point} _point coordinates of mouse in canvas plane
570  */
571 ED.Drawing.prototype.mousemove = function(_point)
572 {
573 	// Only drag if mouse already down and a doodle selected
574 	if (this.mouseDown && this.selectedDoodle != null)
575 	{
576 		// Dragging not started
577 		if (!this.selectedDoodle.isBeingDragged)
578 		{
579 			// Flag start of dragging manoeuvre
580 			this.selectedDoodle.isBeingDragged = true;
581 		}
582 		// Dragging in progress
583 		else
584 		{
585 			// Get mouse position in doodle plane
586 			var mousePosDoodlePlane = this.inverseTransform.transformPoint(_point);
587 			var lastMousePosDoodlePlane = this.inverseTransform.transformPoint(this.lastMousePosition);
588 			
589 			// Get mouse positions in selected doodle's plane
590 			var mousePosSelectedDoodlePlane = this.selectedDoodle.inverseTransform.transformPoint(_point);
591 			var lastMousePosSelectedDoodlePlane = this.selectedDoodle.inverseTransform.transformPoint(this.lastMousePosition);
592 			
593 			// Get mouse positions in canvas plane relative to centre
594 			var mousePosRelCanvasCentre = new ED.Point(_point.x - this.canvas.width/2, _point.y - this.canvas.height/2);
595 			var lastMousePosRelCanvasCentre = new ED.Point(this.lastMousePosition.x - this.canvas.width/2, this.lastMousePosition.y - this.canvas.height/2);
596 				
597 			// Get position of centre of display (canvas plane relative to centre) and of an arbitrary point vertically above
598 			var canvasCentre = new ED.Point(0, 0);
599 			var canvasTop = new ED.Point(0, -100);
600 			
601 			// Get coordinates of origin of doodle in doodle plane
602 			var doodleOrigin = new ED.Point(this.selectedDoodle.originX, this.selectedDoodle.originY);
603 
604 			// Get position of point vertically above doodle origin in doodle plane
605 			var doodleTop = new ED.Point(this.selectedDoodle.originX, this.selectedDoodle.originY - 100);
606 			
607 			// Effect of dragging depends on mode
608 			switch (this.mode)
609 			{
610 				case ED.Mode.None:
611 					break;
612 				case ED.Mode.Move:
613 					// If isMoveable is true, move doodle
614 					if (this.selectedDoodle.isMoveable)
615 					{
616                         this.selectedDoodle.move(mousePosDoodlePlane.x - lastMousePosDoodlePlane.x, mousePosDoodlePlane.y - lastMousePosDoodlePlane.y);
617 					}
618 					// Otherwise rotate it (if isRotatable)
619 					else 
620 					{
621 						if (this.selectedDoodle.isRotatable)
622 						{
623 							// Calculate angles from centre to mouse positions relative to north
624 							var oldAngle = this.innerAngle(canvasTop, canvasCentre, lastMousePosRelCanvasCentre);
625 							var newAngle = this.innerAngle(canvasTop, canvasCentre, mousePosRelCanvasCentre);
626 							
627 							// Work out difference, and change doodle's angle of rotation by this amount
628 							var deltaAngle = newAngle - oldAngle;
629 							this.selectedDoodle.rotation += deltaAngle;
630                             
631                             // Adjust radius property
632                             var oldRadius = Math.sqrt(lastMousePosDoodlePlane.x * lastMousePosDoodlePlane.x + lastMousePosDoodlePlane.y * lastMousePosDoodlePlane.y);
633                             var newRadius = Math.sqrt(mousePosDoodlePlane.x * mousePosDoodlePlane.x + mousePosDoodlePlane.y * mousePosDoodlePlane.y);
634                             this.selectedDoodle.radius += (newRadius - oldRadius);
635                             
636                             // Keep within bounds
637                             this.selectedDoodle.radius = this.selectedDoodle.rangeOfRadius.constrain(this.selectedDoodle.radius);
638  						}
639 					}
640 					break;
641 				case ED.Mode.Scale:
642 					if (this.selectedDoodle.isScaleable)
643 					{
644 						// Get sign of scale (negative scales create horizontal and vertical flips)
645 						var signX = this.selectedDoodle.scaleX/Math.abs(this.selectedDoodle.scaleX);
646 						var signY = this.selectedDoodle.scaleY/Math.abs(this.selectedDoodle.scaleY);
647 
648 						// Calculate change in scale (sign change indicates mouse has moved across central axis)
649 						var changeX = mousePosSelectedDoodlePlane.x/lastMousePosSelectedDoodlePlane.x;
650 						var changeY = mousePosSelectedDoodlePlane.y/lastMousePosSelectedDoodlePlane.y;
651 						
652 						// Ensure scale change is same if not squeezable
653 						if (!this.selectedDoodle.isSqueezable)
654 						{
655 							if (changeX > changeY) changeY = changeX;
656 							else changeY = changeX;
657 						}
658 						
659 						// Check that mouse has not moved from one quadrant to another 
660 						if (changeX > 0 && changeY > 0)
661 						{
662 							// Now do scaling
663 							this.selectedDoodle.scaleX = this.selectedDoodle.scaleX * changeX;
664 							this.selectedDoodle.scaleY = this.selectedDoodle.scaleY * changeY;
665 							
666 							// Constrain scale
667 							this.selectedDoodle.scaleX = this.selectedDoodle.rangeOfScale.constrain(Math.abs(this.selectedDoodle.scaleX)) * signX;
668 							this.selectedDoodle.scaleY = this.selectedDoodle.rangeOfScale.constrain(Math.abs(this.selectedDoodle.scaleY)) * signY;
669 						}
670 						else
671 						{
672 							this.mode = ED.Mode.None;
673 						}
674 					}
675 					break;
676 				case ED.Mode.Arc:
677 
678                     // Calculate angles from centre to mouse positions relative to north
679                     var newAngle = this.innerAngle(doodleTop, doodleOrigin, mousePosSelectedDoodlePlane);
680                     var oldAngle = this.innerAngle(doodleTop, doodleOrigin, lastMousePosSelectedDoodlePlane);
681                     
682                     // Work out difference, and sign of rotation correction
683                     var deltaAngle = newAngle - oldAngle;
684                     if (this.selectedDoodle.isArcSymmetrical) deltaAngle = 2 * deltaAngle;
685                     rotationCorrection = 1;
686 
687                     // Arc left or right depending on which handle is dragging
688                     if (this.selectedDoodle.draggingHandleIndex < 2)
689                     {
690                         deltaAngle =  -deltaAngle;
691                         rotationCorrection = -1;
692                     }
693                     
694                     // Clamp to permitted range and stop dragging if exceeded
695                     if (this.selectedDoodle.rangeOfArc.isBelow(this.selectedDoodle.arc + deltaAngle))
696                     {
697                         deltaAngle = this.selectedDoodle.rangeOfArc.min - this.selectedDoodle.arc;
698                         this.selectedDoodle.arc = this.selectedDoodle.rangeOfArc.min;
699                         this.mode = ED.Mode.None;
700                     }
701                     else if (this.selectedDoodle.rangeOfArc.isAbove(this.selectedDoodle.arc + deltaAngle))
702                     {
703                         deltaAngle = this.selectedDoodle.rangeOfArc.max - this.selectedDoodle.arc;
704                         this.selectedDoodle.arc = this.selectedDoodle.rangeOfArc.max;
705                         this.mode = ED.Mode.None;
706                     }
707                     else
708                     {
709                         this.selectedDoodle.arc += deltaAngle;
710                     }
711                     
712                     // Correct rotation with counter-rotation
713                     if (!this.selectedDoodle.isArcSymmetrical)
714                     {
715                         rotationCorrection = rotationCorrection * deltaAngle/2;
716                         this.selectedDoodle.rotation += rotationCorrection;
717                     }
718 
719 					break;
720 				case ED.Mode.Rotate:
721 					if (this.selectedDoodle.isRotatable)
722 					{
723 						// Calculate angles from centre to mouse positions relative to north
724 						var oldAngle = this.innerAngle(doodleTop, doodleOrigin, lastMousePosDoodlePlane);
725 						var newAngle = this.innerAngle(doodleTop, doodleOrigin, mousePosDoodlePlane);
726 						
727 						// Work out difference, and change doodle's angle of rotation by this amount
728 						var deltaAngle = newAngle - oldAngle;
729 						this.selectedDoodle.rotation = this.selectedDoodle.rotation + deltaAngle;
730 					}
731 					break;
732 				case ED.Mode.Apex:
733 					// Move apex to new position
734 					this.selectedDoodle.apexX += (mousePosSelectedDoodlePlane.x - lastMousePosSelectedDoodlePlane.x);
735 					this.selectedDoodle.apexY += (mousePosSelectedDoodlePlane.y - lastMousePosSelectedDoodlePlane.y);
736 					
737 					// Enforce bounds
738 					this.selectedDoodle.apexX = this.selectedDoodle.rangeOfApexX.constrain(this.selectedDoodle.apexX);
739 					this.selectedDoodle.apexY = this.selectedDoodle.rangeOfApexY.constrain(this.selectedDoodle.apexY);
740 					break;
741                 case ED.Mode.Draw:
742                     var p = new ED.Point(mousePosSelectedDoodlePlane.x,mousePosSelectedDoodlePlane.y);
743                     this.selectedDoodle.addPointToSquiggle(p);
744                     break;
745 				default:
746 					break;		
747 			}
748 			            
749 			// Refresh drawing
750 			this.repaint();				
751 		}
752 		
753 		// Store mouse position
754 		this.lastMousePosition = _point;
755 	}
756 }

1555 ED.Report = function(_drawing)
