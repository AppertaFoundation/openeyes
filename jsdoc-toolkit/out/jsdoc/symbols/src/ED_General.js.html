 89 /**
 90  * Draws doodle or performs a hit test if a Point parameter is passed
 91  *
 92  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
 93  */
 94 ED.Freehand.prototype.draw = function(_point)
 95 {
 96 	// Get context
 97 	var ctx = this.drawing.context;
 98 	
 99 	// Call draw method in superclass
100 	ED.Freehand.superclass.draw.call(this, _point);
101 	
102 	// Boundary path
103 	ctx.beginPath();
104 	
105 	// Freehand
106 	ctx.rect(-150, -150, 300, 300);
107 	
108 	// Close path
109 	ctx.closePath();
110 	
111 	// Set line attributes
112 	ctx.lineWidth = 2;
113     this.isFilled = false;
114     ctx.strokeStyle = "rgba(0, 0, 0, 0)";
115     if (this.isSelected) ctx.strokeStyle = "gray";
116     if (this.isForDrawing) ctx.strokeStyle = "blue";
117 
118 	// Draw boundary path (also hit testing)
119 	this.drawBoundary(_point);
120 	
121 	// Other stuff here
122 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
123 	{
124         // Iterate through squiggles, drawing them
125         for (var i = 0; i < this.squiggleArray.length; i++)
126         {
127             var squiggle = this.squiggleArray[i];
128             
129             ctx.beginPath();
130             
131             // Squiggle attributes
132             ctx.lineWidth = squiggle.thickness;
133             ctx.strokeStyle = squiggle.colour.rgba();
134             ctx.fillStyle = squiggle.colour.rgba();
135             
136             // Iterate through squiggle points
137             for (var j = 0; j < squiggle.pointsArray.length; j++)
138             {
139                 ctx.lineTo(squiggle.pointsArray[j].x, squiggle.pointsArray[j].y);
140             }
141             
142             // Draw squiggle
143             ctx.stroke();
144             
145             // Optionally fill if squiggle is complete (stops filling while drawing)
146             if (squiggle.filled && squiggle.complete) ctx.fill();
147         }
148 	}
149 	
150 	// Coordinates of handles (in canvas plane)
151 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(150, -150));
152 	
153 	// Draw handles if selected but not if for drawing
154 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
155 	
156 	// Return value indicating successful hittest
157 	return this.isClicked;
158 }

240 /**
241  * Draws doodle or performs a hit test if a Point parameter is passed
242  *
243  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
244  */
245 ED.Lable.prototype.draw = function(_point)
246 {
247 	// Get context
248 	var ctx = this.drawing.context;
249 	
250 	// Call draw method in superclass
251 	ED.Lable.superclass.draw.call(this, _point);
252     
253     // Set font
254     ctx.font = this.lableFont;
255     
256     // Calculate pixel width of text with padding
257     this.lableWidth = ctx.measureText(this.lableText).width + this.padding * 2;
258 	
259 	// Boundary path
260 	ctx.beginPath();
261 	
262 	// Lable boundary
263 	ctx.rect(-this.lableWidth/2, -this.lableHeight/2, this.lableWidth, this.lableHeight);
264     
265 	// Close path
266 	ctx.closePath();
267 	
268 	// Set line attributes
269 	ctx.lineWidth = 2;
270     this.isFilled = false;
271     ctx.strokeStyle = "rgba(0, 0, 0, 0)";
272     if (this.isSelected) ctx.strokeStyle = "gray";
273     
274 	// Draw boundary path (also hit testing)
275 	this.drawBoundary(_point);
276 	
277 	// Other stuff here
278 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
279 	{    
280         // Draw boundary
281         ctx.stroke();
282         
283         // Draw text
284         ctx.fillText(this.lableText, -this.lableWidth/2 + this.padding, this.lableHeight/6);
285     }
286 	
287 	// Coordinates of handles (in canvas plane)
288 	this.handleArray[2].location = this.transform.transformPoint(new ED.Point(this.lableWidth/2, -this.lableHeight/2));
289 	
290 	// Draw handles if selected but not if for drawing
291 	if (this.isSelected && !this.isForDrawing) this.drawHandles(_point);
292 	
293 	// Return value indicating successful hittest
294 	return this.isClicked;
295 }

301  */
302 ED.Lable.prototype.addLetter = function(_keyCode)
303 {
304     // need code here to convert to character

307     if (this.lableText.length < this.maximumLength) this.lableText += character;

498 /**
499  * Draws doodle or performs a hit test if a Point parameter is passed
500  *
501  * @param {Point} _point Optional point in canvas plane, passed if performing hit test
502  */
503 ED.Slider.prototype.draw = function(_point)
504 {
505 	// Get context
506 	var ctx = this.drawing.context;
507 	
508 	// Call draw method in superclass
509 	ED.Slider.superclass.draw.call(this, _point);
510 	
511 	// Boundary path
512 	ctx.beginPath();
513     
514     // Outline of slider
515     ctx.moveTo(-this.halfLength, -this.halfHeight);
516     ctx.lineTo(this.halfLength, -this.halfHeight);
517     ctx.arc(this.halfLength, 0, this.halfHeight, -Math.PI/2, Math.PI/2, false);
518     ctx.lineTo(-this.halfLength, this.halfHeight);    
519     ctx.arc(-this.halfLength, 0, this.halfHeight, Math.PI/2, -Math.PI/2, false);
520     ctx.closePath();
521     
522     // Colour of fill
523     
524     // Colors for gradient
525     var topColour = "rgba(160, 160, 160, 1)";
526     var bottomColour = "rgba(240, 240, 240, 1)";
527     
528     // Vertical gradient
529     var gradient = ctx.createLinearGradient(0, -this.halfHeight, 0, this.halfHeight)
530     gradient.addColorStop(0, topColour);
531     gradient.addColorStop(1, bottomColour);
532     ctx.fillStyle = gradient;
533     
534 	// Set line attributes
535 	ctx.lineWidth = 2;
536     
537     // Colour of outer line is dark gray
538     ctx.strokeStyle = "rgba(120,120,120,0.8)";
539 	
540 	// Draw boundary path (also hit testing)
541 	this.drawBoundary(_point);
542 	
543 	// Other stuff here
544 	if (this.drawFunctionMode == ED.drawFunctionMode.Draw)
545 	{
546 	}
547 	
548     // Coordinates of handles (in canvas plane)
549     this.handleArray[4].location = this.transform.transformPoint(new ED.Point(this.apexX, this.apexY));
550     
551 	// Draw handles if selected
